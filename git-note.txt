git status = checking file changes
git init = create git repository
git add . = add all files to repository 
git commit -m "example commit" = description of changes
git log = view commits
git log --oneline = view commits with one line view
git branch exampleBranchName = create new branch
git checkout exampleBranchName = switch branch
git branch -b exampleBranchName = create + switch branch
git branch -d exampleBranchName = delete branch
git push -u origin main = push into github
git pull 'remote_name' 'branch_name' = pull  from github
git pull origin main --allow-unrelated-histories = pull from github repo
